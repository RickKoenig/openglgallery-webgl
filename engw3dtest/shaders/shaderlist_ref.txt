// this is a reference to the D3D windows vertex types
// vertex_types
/*
VT   pos uv						// plane texture

VCT  pos col uv					// cvert lighting method
VT2  pos uv0 uv1				// lightmaps, 2 sets of uv's, better resolution than cvert method 'lightmap userprop'

VNT  pos norm uv				// normal lighting method
VNTM pos norm tangent binorm uv // bump map (3 direction vectors, instead of just the 1 normal)
VN   pos norm					// (no uvs) good for 'stan' shader

VNT2 pos norm uv0 uv1			// first attempt at toon shader, failed
VE	 pos0 pos1					// second attempt at toon shader, position and position at other side of edge,  generated by calling 'normalfy' or 'edgeify' function on a model (not used by artists)
*/

// shadername shader.fx vertex_type
// try .fxo	or .fx
	tex tex.fx VCT							// no lighting // // can be VT or VCT see u_s_3dtest.cpp for why
	texadd texadd.fx VCT					// no lighting // // can be VT or VCT see u_s_3dtest.cpp for why
	lightmap lightmap.fx VT2				// 2 textures, 2 uvs
// try .fxo	or .fx
	cvert cvert.fx VCT						// colored verts
	vdiff vdiff.fx VNT						// vertex diffuse lighting
	pdiff pdiff.fx VNT						// pixel diffuse lighting
	vdiffspec vdiffspec.fx VNT				// vertex amb/diff/spec lighting
	pdiffspec pdiffspec.fx VNT				// pixel amb/diff/spec lighting and with 2 textures with common uvs

	pnormalmap pnormalmap.fx VNTM			// main texture and a normal map with amb/diff/spec
	
	makeshadowmap makeshadowmap.fx VNT		// shadow map maker
	useshadowmap useshadowmap.fx VNT		// shadow map user
	shadowmapviewer shadowmapviewer.fx VT	// no lighting // can be VT
	
	env env.fx VNT							// environment mapper
	
	tex2c tex2c.fx VCT						// uses 2 matcolors, nonzero tex gets main color, zero gets alt color, for fonts


	stan stan.fx VNT						// the stan effect, see monkey island
	
	toon toon.fx VNT						// toon effect
	toon2 toon2.fx VNT						// toon effect milder edges
	edge edge.fx VE 						// better toon effect, vertex shader edge drawer 
	edge2 edge2.fx VE 						// better toon effect, vertex shader edge drawer milder

	envv2 envv2.fx VNT						// env mapper, uses global env tex (v2)
	
	texvel texvel.fx VNT					// tex with texvel
	envv2texvel envv2texvel.fx VNT			// env mapper, uses global env tex (v2)
	
	
	laserbeam laserbeam.fx VNT				// try a special laserbeam shader ( experimental)
	cvertvel cvertvel.fx VCT				// colored verts texvel
